# React 泛用節流

## 定義類型

```
//定義值和改變值的函數 ValueProps<T>
type ValueProps<T> = {
  value: T;
  onChange: (arg: T) => void;
};

type PropsType<T> = {
    //接受一個子組件，並且可以接受屬性
    children: (props: ValueProps<T>) => ReactElement<ValueProps<T>>;//
    //傳遞給子組件的屬性
    childrenProps: ValueProps<T>;
    //等待時間
    wait?: number;
    //回呼函數
    action?: (value: T) => void;
};
```

## 節流高階組件

```
const DebounceHOC = <T,>({ children, childrenProps, action, wait = 1000 }: PropsType<T>) => {
    const { value, onChange } = childrenProps;
    //設定節流的初始值
    const [debounceValue, setDebounceValue] = useState<T>(value);

    useEffect(() => {
        //在父層變化時更新節流的值
        setDebounceValue(value);
    }, [value]);

    useEffect(() => {
        //避免再父層變化時也執行節流
        if (value === debounceValue) return;
        //基本節流
        const handler = setTimeout(() => {
          onChange(debounceValue);
          if (!action) return;
          action(debounceValue);
        }, wait);
        //清除節流
        return () => clearTimeout(handler);
    }, [debounceValue]);

    //子組件限制
    if (!children || typeof children !== 'function') {
        throw Error('DebounceHOC must have children');
    }

    return children({ value: debounceValue, onChange: setDebounceValue });
};

export default DebounceHOC;
```

## 使用方式

```
export const Example = () => {
    const [value, setValue] = useState<string>('');

    function onChange(newValue: string) {
        setValue(newValue);
    }

    function action(newValue: string) {
        console.log('DebounceHOC Example', newValue);
    }
    return (
        <Fragment>
            <div>value: {value}</div>
            <DebounceHOC<string> childrenProps={{ value, onChange }} action={action}>
            {({ value, onChange }) => <input value={value} onChange={e => onChange(e.target.value)} />}
            </DebounceHOC>
        </Fragment>
    );
};
```
