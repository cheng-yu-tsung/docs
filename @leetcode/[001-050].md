> ## 兩數之和

```Typescript
//nums = [2,7,11,15], target = 9 =>[0,1]
function twoSum(nums: number[], target: number): number[] | void {
  let d = {};
  for (let i = 0; i < nums.length; ++i) {
    const x = nums[i];
    const y = target - x;
    if (d[y] !== undefined) {
      return [d[y], i];
    }
    d[x] = i;
  }
}
```
